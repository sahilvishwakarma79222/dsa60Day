	static private int second(int[] nums,int n) {
		if(n<2) {
			return -1;
		}
		int large=Integer.MIN_VALUE;
		int second_large=Integer.MIN_VALUE;
		for(int i=0;i<nums.length;i++) {
			if(nums[i]>large) {
				second_large=large;
				large=nums[i];
			}
			if(nums[i]>second_large && nums[i]<large) {
				second_large=nums[i];
			}
		}
	    return (second_large == Integer.MIN_VALUE) ? -1 : second_large; // i can use if condition also in this condition
		
	}

	
	public static void main(String[] args) {
		int[] nums = {5, 5, 5, 5, 5};
		System.out.println(second(nums, nums.length));
		
	}
